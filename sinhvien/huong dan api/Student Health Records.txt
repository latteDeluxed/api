    GET /health_records.php - Get all health records (admin view)

    GET /health_records.php?record_id=X - Get single health record

    GET /health_records.php?student_id=X - Get all health records for a student

    GET /health_records.php?record_type=X - Filter by record type (Medical/Dental/Vaccination/Other)

    GET /health_records.php?start_date=YYYY-MM-DD&end_date=YYYY-MM-DD - Filter by date range

    POST /health_records.php - Create new health record

    PUT /health_records.php?record_id=X - Update health record

    DELETE /health_records.php?record_id=X - Delete health record
