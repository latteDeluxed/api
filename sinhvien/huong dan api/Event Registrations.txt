    GET /event_registrations.php?registration_id=X - Get single registration

    GET /event_registrations.php?event_id=X - Get all registrations for an event

    GET /event_registrations.php?event_id=X&student_id=Y - Check if student is registered

    GET /event_registrations.php?student_id=X - Get all event registrations for a student

    GET /event_registrations.php?status=X - Filter by attendance status

    PUT /event_registrations.php?registration_id=X - Update attendance status

    DELETE /event_registrations.php?registration_id=X - Cancel registration
